    let result = "E";
    if (goalsFor > goalsAgainst) result = "G";
    if (goalsFor < goalsAgainst) result = "P";

    return {
      vs: opponent,
      local: isHome,
      resultado: result,
      marcador: `${goalsFor}-${goalsAgainst}`,
    };
  };
}
